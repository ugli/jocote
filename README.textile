h3. Jocote - JMS made easy

<pre><code>
import static org.hamcrest.CoreMatchers.*;
import static org.junit.Assert.*;
import se.ugli.jocote.*;

DriverManager.register("se.ugli.jocote.activemq.ActiveMqDriver");
Connection connection = DriverManager.getConnection("activemq:/TEST");  
connection.put("hello world");
String message = connection.get();
assertThat(message, equalTo("hello world"));
connection.close();
